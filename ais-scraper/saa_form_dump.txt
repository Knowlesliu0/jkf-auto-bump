        //倒數計時器  
        //countdownId:剩餘時間Id
        //countdown:倒數時間
        //statusLabelId:出價狀況Id
        function timerCountDown(countdownId, countdown, statusLabelId,stop_yn) {
      
            var times = countdown;
            countdown--;
            if (countdown < 0) {
              
                
                document.getElementById(countdownId).innerHTML = "00:00:00";
                statusLabelId.innerHTML = "拍賣結束";  
                return;
            } else {

                var result = '';
                var h = Math.floor(times / (60 * 60));
                if (h < 10)
                    result += '0' + h + '：';
                else
                    result += h + '：';
                times = times - h * (60 * 60);

                var m = Math.floor(times / (60));
                if (m < 10)
                    result += '0' + m + '：';
                else
                    result += m + '：';
                times = times - (m * 60);

                var s = times;
                if (s < 10)
                    result += '0' + s;
                else
                    result += s;

                //countdownId.innerHTML = result;
                document.getElementById(countdownId).innerHTML = result ;
                statusLabelId.innerHTML = "拍賣中";
                if (stop_yn == "Y")
                    statusLabelId.innerHTML = "停拍";

                if (countdown <= 300)
                    statusLabelId.innerHTML = "即將結束";


                window.setTimeout("timerCountDown('" + countdownId + "'," + countdown + "," + statusLabelId.id + ",'" + stop_yn + "')", 1000);
            }
        }
